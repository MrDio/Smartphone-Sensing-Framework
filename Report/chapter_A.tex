\begin{definition}
A function $f : \mathbb{R} \to \mathbb{R}$ is \emph{odd}, if $f(-x)= -f(x)$ holds for all $x \in \mathbb{R}$.
\end{definition}

\begin{theorem}
If $f$ is an odd function and $a > 0$, then
\[
\int_{-a}^a f(x) \,dx = 0\,.
\]
\end{theorem}
